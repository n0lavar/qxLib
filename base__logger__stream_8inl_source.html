147 inline void base_logger_stream::format_line(
148  logger_buffers<char_type>& buffers,
149  log_level eLogLevel,
150  const char_type* pszFormat,
151  const char_type* pszCategory,

173  break;
174 
175  case log_level::warning:
176  buffers.sFormat = QX_STR_PREFIX(char_type, "[W][") + buffers.sFormat;
177  break;
178 
179  case log_level::error:
180  buffers.sFormat = QX_STR_PREFIX(char_type, "[E][") + buffers.sFormat;
181  break;
182 
183  case log_level::critical:
184  buffers.sFormat = QX_STR_PREFIX(char_type, "[C][") + buffers.sFormat;
185  break;
186 
187  default:
188  buffers.sFormat = QX_STR_PREFIX(char_type, " [") + buffers.sFormat;
189  break;
190  }
191 
192  constexpr auto pszStringFormatSpecifier = get_format_specifier<char_type, const char_type*>();
193  buffers.sFormat += QX_STR_PREFIX(char_type, "][");
194  buffers.sFormat += pszStringFormatSpecifier;
195  if (pszCategory)
196  buffers.sFormat += QX_STR_PREFIX(char_type, "][");
197  buffers.sFormat += pszStringFormatSpecifier;
198  buffers.sFormat += QX_STR_PREFIX(char_type, "::");
199  buffers.sFormat += pszStringFormatSpecifier;
200  buffers.sFormat += QX_STR_PREFIX(char_type, "::%d] ");
201  buffers.sFormat += buffers.sMessage;
202  buffers.sMessage.sprintf(buffers.sFormat.c_str(), pszCategory ? pszCategory : "", pszFile, pszFunction, nLine);
203  buffers.sMessage += QX_CHAR_PREFIX(char_type, '\n');
