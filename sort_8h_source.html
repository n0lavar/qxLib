330  @brief Quick sort based on three-way partitioning
331  @details Unstable
332  @complexity O(Nlog(N))
333  @tparam RandomIt - random access iterator
334  @tparam Compare - function object type satisfying binary predicate requirements
335  @param begin - begin iterator
336  @param end - end iterator
337  @param compare - comparison function
338 **/
339 template<typename RandomIt, typename Compare = std::less<>>
340 inline void sort_quick_three_way(RandomIt begin, RandomIt end, Compare compare = Compare())
341 {
342  if (!sort_required(begin, end))
343  return;
344 
345  const i64 nRight = (end - begin) - 1;
346  i64 nLeft = 0;
347 
348  if (nRight > 0)
349  {
350  auto pivot = *(begin + nLeft + (nRight - nLeft) / 2);
351 
352  i64 i = nLeft;
353  i64 j = nRight;
354 
355  i64 nIndex1 = nLeft - 1;
356  i64 nIndex2 = nRight + 1;
357 
358  while (i <= j)
359  {
360  while (compare(*(begin + i), pivot))
361  ++i;
362 
363  while (compare(pivot, *(begin + j)))
364  --j;
365 
366  if (i < j)
367  {
368  if (compare(pivot, *(begin + i)))
369  {
370  if (compare(*(begin + j), pivot))
371  {
372  // nLeft[i] != nPivot and nLeft[j] != nPivot
373  std::swap(*(begin + i), *(begin + j));
374  }
375  else
376  {
377  // nLeft[i] != nPivot and nLeft[j] == nPivot
378  std::swap(*(begin + i), *(begin + j));
379  std::swap(*(begin + ++nIndex1), *(begin + i));
380  }
381  }
382  else if (compare(*(begin + j), pivot))
383  {
384  // nLeft[i] == nPivot nLeft[j] != nPivot
385  std::iter_swap(begin + i, begin + j);
386  std::iter_swap(begin + j, begin + --nIndex2);
387  }
388  else
389  {
390  // nLeft[i] == nPivot and nLeft[j] == nPivot
391  std::iter_swap(begin + ++nIndex1, begin + i);
392  std::iter_swap(begin + j, begin + --nIndex2);
393  }
394 
395  ++i;
396  --j;
397  }
398  else if (i == j)
399  {
400  j = i - 1;
401  ++i;
402  break;
403  }
404  }
405 
406  for (i64 k = nLeft; k <= nIndex1; ++k)
407  std::iter_swap(begin + k, begin + j--);
408 
409  for (i64 k = nRight; k >= nIndex2; --k)
410  std::iter_swap(begin + i++, begin + k);
411 
412  sort_quick_three_way(begin, begin + (j + 1), compare);
413  sort_quick_three_way(begin + i, end, compare);
414  }
415 }

433  @brief Quick sort based on dual-pivot partitioning
434  @details Unstable

436  @tparam RandomIt - random access iterator

444 {

448  i64 nRight = (end - begin) - 1;
449  i64 nLeft = 0;
450 
451  if (nRight > nLeft)
452  {

454  std::iter_swap(begin + nRight - (nRight - nLeft) / 3, begin + nRight);
455 
456  if (compare(*(begin + nRight), *(begin + nLeft)))
457  std::iter_swap(begin + nLeft, begin + nRight);
458 
459  if (nRight - nLeft == 1)
460  return;
461 
462  auto pivot1 = *(begin + nLeft);
463  auto pivot2 = *(begin + nRight);
464 
465  if (compare(pivot1, pivot2))
466  {
467  i64 nLess = nLeft + 1;
468  i64 nGreater = nRight - 1;
469 
470  while (!compare(*(begin + nGreater), pivot2))
471  --nGreater;
472 
473  while (!compare(pivot1, *(begin + nLess)))
474  ++nLess;

480  {

482  }

484  {
485  std::iter_swap(begin + k, begin + nGreater--);
486  while (!compare(*(begin + nGreater), pivot2))
487  --nGreater;

491  }

493  k++;

497  std::iter_swap(begin + nGreater + 1, begin + nRight);
498

500  sort_quick_dual_pivot(begin + nGreater + 2, end, compare);
501  sort_quick_dual_pivot(begin + nLess, begin + nGreater + 1, compare);
502  }
503  else
504  {
505  // pivot1 == pivot2
506  i64 nLess = nLeft + 1;
507  i64 nGreater = nRight - 1;
508 
509  while (compare(pivot1, *(begin + nGreater)))
510  --nGreater;
511 
512  while (compare(*(begin + nLess), pivot1))
513  ++nLess;
514 
515  i64 k = nLess;
516  while (k <= nGreater)
517  {
518  if (compare(*(begin + k), pivot1))

520  std::iter_swap(begin + k, begin + nLess++);
521  }

559 /**
560  @brief Merge with temp vector
561  @complexity O(N), O(N) memory
562  @tparam RandomIt - random access iterator
563  @tparam Compare - function object type satisfying binary predicate requirements
564  @param begin - begin iterator
565  @param end - end iterator
566  @param compare - comparison function
567  @param pPreallocatedBuffer - preallocated buffer. default value will allocate a new one
568 **/
569 template<typename RandomIt, typename Compare = std::less<>>
570 inline void merge(
571  RandomIt begin,
572  RandomIt end,
573  Compare compare = Compare(),
574  vector_of_values<RandomIt>* pPreallocatedBuffer = nullptr)
575 {
576  using iter_diff = decltype(RandomIt() - RandomIt());
577 
578  u64 nInd1 = 0;
579  u64 nInd2 = 0;
580  RandomIt mid = begin + (end - begin) / 2;
581 
582  const bool bCreateBuffer = !pPreallocatedBuffer;
583  if (bCreateBuffer)
584  pPreallocatedBuffer = new vector_of_values<RandomIt>;
585 
586  pPreallocatedBuffer->resize(static_cast<size_t>(end - begin));
587 
588  // merge
589  while ((begin + static_cast<iter_diff>(nInd1) < mid) && (mid + static_cast<iter_diff>(nInd2) < end))
590  {
591  if (compare(*(begin + static_cast<iter_diff>(nInd1)), *(mid + static_cast<iter_diff>(nInd2))))
592  {
593  (*pPreallocatedBuffer)[nInd1 + nInd2] = *(begin + static_cast<iter_diff>(nInd1));
594 
595  nInd1++;
596  }
597  else
598  {
599  (*pPreallocatedBuffer)[nInd1 + nInd2] = *(mid + static_cast<iter_diff>(nInd2));
600 
601  nInd2++;
602  }
603  }
604 
605  // append tails
606  while (begin + static_cast<iter_diff>(nInd1) < mid)
607  {
608  (*pPreallocatedBuffer)[nInd1 + nInd2] = *(begin + static_cast<iter_diff>(nInd1));
609 
610  nInd1++;
611  }
612 
613  while (mid + static_cast<iter_diff>(nInd2) < end)
614  {
615  (*pPreallocatedBuffer)[nInd1 + nInd2] = *(mid + static_cast<iter_diff>(nInd2));
616 
617  nInd2++;
618  }
619 
620  // copy to source
621  for (auto it = begin; it < end; ++it)
622  *it = (*pPreallocatedBuffer)[static_cast<size_t>(it - begin)];
623 
624  if (bCreateBuffer)
625  delete pPreallocatedBuffer;
626 }

691  @param end - end iterator
