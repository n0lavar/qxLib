147 /**
148  @brief Integrate using probabilistic algorithm Monte Carlo
149  @param funcIsInside - func that returns
150  1 if point is inside shape with positive value
151  0 if point is not inside shape
152  -1 if point is inside shape with negative value
153  @param pos0 - left down corner coordinates
154  @param pos1 - right up corner coordinates
155  @param nPointsPerOneSquare - points per 1 square (more is better)
156  @tparam Function2dType - function that takes double and returns double
157  @retval - approximate integral
158 **/
159 template<class Function2dType>
160 inline double integrate_monte_carlo(
161  const Function2dType& funcIsInside,
162  glm::dvec2 pos0,
163  glm::dvec2 pos1,
164  size_t nPointsPerOneSquare = 1000)
165 {
166  const double fArea = std::abs(pos1.x - pos0.x) * std::abs(pos1.y - pos0.y);
167  const int nTotalPoints = static_cast<int>(std::ceil(fArea * static_cast<double>(nPointsPerOneSquare)));
168 
169  int points_inside = 0;
170 
171  std::default_random_engine generator(static_cast<unsigned>(std::time(nullptr)));
172 
173  std::uniform_real_distribution<double> x_dist(pos0.x, pos1.x);
174  std::uniform_real_distribution<double> y_dist(pos0.y, pos1.y);
175 
176  for (int i = 0; i < nTotalPoints; ++i)
177  {
178  int f = funcIsInside(x_dist(generator), y_dist(generator));
179  if (f > 0)
180  points_inside++;
181  else if (f < 0)
182  points_inside--;
183  }
184 
185  return (static_cast<double>(points_inside) / nTotalPoints) * fArea;
186 }
