115 /**
116  @brief Greatest common divisor
117  @details Euclid's algorithm
118  based on fact gcd(A, B) == gcd(B, A mod B)
119  @complexity O(log(second))
120  @param nFirst - first num
121  @param nSecond - second num
122  @retval - greatest common divisor if first and second > 0, otherwise 0
123 **/
124 constexpr int gcd(int nFirst, int nSecond)
125 {
126  if (nFirst == 0 || nSecond == 0)
127  return 0;
128 
129  while (nSecond != 0)
130  {
131  const int nRemainder = nFirst % nSecond;
132  nFirst = nSecond;
133  nSecond = nRemainder;
134  }
135 
136  return abs(nFirst);
137 }

157 /**
158  @brief Power function for integer power
159  @details About 2.22 times (positive powers)
160  1.7 times (positive and negative powers)
161  2.33 times (negative powers)
162  faster then std::pow
163  @complexity O(log(power))
164  @tparam T - Integral or floating point type
165  @param number - integral of floating point value
166  @param nPower - integral power
167  @retval - number ^ power
168 **/
169 template<typename T>
170 inline double pow(T number, int nPower)
171 {
172  static_assert(std::is_integral_v<T> || std::is_floating_point_v<T>, "Integral or floating point required");
173 
174  const bool bNegativePower = nPower < 0;
175  const size_t nPositivePower = static_cast<size_t>(std::abs(nPower));
176 
177  double fResult = 1.0;
178  switch (nPositivePower)
179  {
180  case 0:
181  break;
182 
183  case 1:
184  fResult = static_cast<double>(number);
185  break;
186 
187  case 2:
188  fResult = static_cast<double>(number * number);
189  break;
190 
191  default:
192  const std::bitset<std::numeric_limits<int>::digits> powerBitSet(nPositivePower);
193 
194  std::array<double, std::numeric_limits<int>::digits> powers;
195 
196  powers[0] = static_cast<double>(number);
197 
198  size_t nCurPower = 1;
199  size_t nCurIndex = 1;
200 
201  while (nCurPower < nPositivePower)
202  {
203  powers[nCurIndex] = powers[nCurIndex - 1] * powers[nCurIndex - 1];
204  nCurPower *= 2;
205  nCurIndex++;
206  }
207 
208  for (size_t i = 0; i < nCurIndex; ++i)
209  if (powerBitSet.test(i))
210  fResult *= powers[i];
211 
212  break;
213  }
214 
215  return bNegativePower ? 1.0 / fResult : fResult;
216 }

246 /**
247  @brief Checks if value is between left and right
248  @tparam T - value type
249  @tparam Compare - comparator type
250  @param left - left value
251  @param value - value
252  @param right - right value
253  @param compare - comparator function
254  @retval - true, left <= value <= right
255 **/
256 template<typename T, typename Compare = std::less_equal<>>
257 constexpr bool between(T left, T value, T right, Compare compare)
258 {
259  if constexpr (std::is_enum_v<T>)
260  {
261  i64 l = static_cast<i64>(left);
262  i64 r = static_cast<i64>(right);
263  i64 v = static_cast<i64>(value);
264  return between(l, v, r, compare);
265  }
266  else if constexpr (std::is_integral_v<T> && std::is_same_v<Compare, std::less_equal<>>)
267  {
268  return compare(static_cast<size_t>(value - left), right - left);
269  }
270  else if constexpr (std::is_floating_point_v<T> && std::is_same_v<Compare, std::less_equal<>>)
271  {
272  return epsilon_less_equal(left, value) && epsilon_less_equal(value, right);
273  }
274  else
275  {
276  return compare(left, value) && compare(value, right);
277  }
278 }
