auto calc_fibonacci = qx::make_recursive_lambda(
[](const auto& calc_fibonacci, int n) -> int // return value is required
{
if (n == 1 || n == 2)
return 1;
if (n == 0)
return 0;
return calc_fibonacci(n - 1) + calc_fibonacci(n - 2);
});
int nResult = calc_fibonacci(6);
