283 template<class TObserver>
284 inline void subject<TObserver>::detach(void* pObserver) noexcept
285 {
286  if (m_nIterators == 0)
287  {
288  std::erase_if(
289  m_Tokens,
290  [pObserver](const observer_token_data* pToken)
291  {
292  return pToken->m_pObserver == pObserver;
293  });
294 
295  m_Observers.erase(
296  std::remove(m_Observers.begin(), m_Observers.end(), pObserver),
297  m_Observers.end());
298  }
299  else
300  {
301  std::replace_if(
302  m_Tokens.begin(),
303  m_Tokens.end(),
304  [pObserver](const observer_token_data* pToken)
305  {
306  return pToken && pToken->m_pObserver == pObserver;
307  },
308  static_cast<observer_token_data*>(nullptr));
309 
310  std::replace(
311  m_Observers.begin(),
312  m_Observers.end(),
313  static_cast<TObserver*>(pObserver),
314  static_cast<TObserver*>(nullptr));
315  }
316 }
