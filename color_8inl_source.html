296 inline std::optional<color> color::from_string(string_view svColorName) noexcept
297 {
298  if (const auto optColor = detail::string_to_color_converter::get_instance().get(string_hash(svColorName)))
299  return *optColor;
300 
301  std::optional<color> optColor;
302 
303  const bool bStartsWith0x =
304  svColorName.starts_with(QX_TEXT("0x")) && (svColorName.size() == 8 || svColorName.size() == 10);
305 
306  if (bStartsWith0x || svColorName.starts_with(QX_TEXT("#")) && svColorName.size() == 7)
307  {
308  // TODO no conversion method for string view so far
309  // but with SSO this should be fine
310  const size_t nOffset = bStartsWith0x ? 2 : 1;
311  string sColorName = string_view(svColorName.data() + nOffset, svColorName.size() - nOffset);
312 
313  auto ReadHex = [](string& s) -> std::optional<color>
314  {
315  s.to_lower();
316  if (const auto optInt = s.template to<u64>(QX_TEXT("%llx")))
317  return color(*optInt);
318 
319  return std::nullopt;
320  };
321 
322  if (sColorName.length() == 6)
323  {
324  sColorName += QX_TEXT("FF");
325  optColor = ReadHex(sColorName);
326  }
327  else if (sColorName.length() == 8)
328  {
329  optColor = ReadHex(sColorName);
330  }
331  }
332 
333  return optColor;
334 }
