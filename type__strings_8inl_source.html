147 template<class T, class char_t>
148 constexpr auto type_strings<T, char_t>::get_template_parameters()
149 {
150  std::array<string_view_type, get_num_template_parameters()> tokens;
151 
152  const string_view_type svSignature = create_signature();
153 
154  size_t nTokenStart = 0;
155  size_t nBraceCounter = 0;
156  size_t nParam = 0;
157 
158  const auto add_token = [&tokens, &svSignature](size_t nToken, size_t nStart, size_t nEnd)
159  {
160  while (svSignature[nStart] == QX_CHAR_PREFIX(char_t, ' '))
161  ++nStart;
162 
163  while (svSignature[nEnd] == QX_CHAR_PREFIX(char_t, ' '))
164  --nEnd;
165 
166  constexpr string_view_type classMarker = QX_STR_PREFIX(char_t, "class ");
167  if (string_view_type(svSignature.data() + nStart, nEnd - nStart).starts_with(classMarker))
168  nStart += classMarker.size();
169 
170  tokens[nToken] = string_view_type(svSignature.data() + nStart, nEnd + 1 - nStart);
171  };
172 
173  bool bLambda = false;
174 
175  for (size_t i = 0; i < svSignature.size(); ++i)
176  {
177  switch (svSignature[i])
178  {
179  case QX_CHAR_PREFIX(char_t, '<'):
180  ++nBraceCounter;
181 
182  if (nTokenStart == 0)
183  nTokenStart = i + 1;
184  else if (!bLambda && svSignature.find(lambdaMarker, i + 1) == i + 1)
185  bLambda = true;
186 
187  break;
188 
189  case QX_CHAR_PREFIX(char_t, '>'):
190  --nBraceCounter;
191  if (bLambda)
192  bLambda = false;
193 
194  break;
195 
196  case QX_CHAR_PREFIX(char_t, ','):
197  if (nBraceCounter == 1)
198  {
199  add_token(nParam, nTokenStart, i - 1);
200  nTokenStart = i + 1;
201  ++nParam;
202  }
203  break;
204  }
205  }
206 
207  if (tokens.size() > 0)
208  {
209  size_t nTokenEnd = svSignature.size() - 1;
210 
211  if (svSignature[nTokenEnd] == QX_CHAR_PREFIX(char_t, ' '))
212  --nTokenEnd;
213 
214  if (svSignature[nTokenEnd] == QX_CHAR_PREFIX(char_t, '>'))
215  --nTokenEnd;
216 
217  add_token(nParam, nTokenStart, nTokenEnd);
218  }
219 
220  return tokens;
221 }
