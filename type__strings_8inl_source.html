128  while (svSignature[nEnd] == ' ')

133  nStart += sizeof(classMarker) - 1;

136  };

141  {
142  switch (svSignature[i])
143  {
144  case '<':
145  ++nBraceCounter;
146 
147  if (nTokenStart == 0)

149  else if (!bLambda && svSignature.find(lambdaMarker, i + 1) == i + 1)
150  bLambda = true;
151

154  case '>':
