280  /**
281  @brief Get or add class data of a given type
282  @tparam component_t - final class type
283  @tparam callable_t - a callable that takes a class data
284  @param iterateClassDataFunction - a function that iterates over a class data from base_component_t + 1 to component_t
285  @retval - or add class data
286  **/
287  template<
288  std::derived_from<base_component_t> component_t,
289  callable_c<void, class_data&> callable_t = decltype(stub_callback)>
290  class_data& get_or_add_class_data(callable_t iterateClassDataFunction = stub_callback) noexcept
291  {
292  /*
293  | | rtti_base | c1 | base_component_t | c2 | c3 | component_t |
294  | t1 | <=======================================================> |
295  | t2 | <===============================> |
296  | t3 | <====================> |
297  */
298  using t1 = typename component_t::inheritance_tuple_type;
299  using t2 = typename base_component_t::inheritance_tuple_type;
300  using t3 = tuple::remove_t<t1, t2>;
301 
302  class_data* pClassData = &m_RootClass;
303  iterateClassDataFunction(*pClassData);
304 
305  tuple::iterate<t3>(
306  [&pClassData, &iterateClassDataFunction]<class T, size_t I>()
307  {
308  pClassData = &pClassData->get_or_add_class_data(T::get_class_id_static());
309  iterateClassDataFunction(*pClassData);
310  });
311 
312  return *pClassData;
313  }

330  /**

332  @tparam callable_t - a callable that takes a class data
333  @param pRawComponent - an object used to identify a class type
334  @param iterateClassDataFunction - a function that iterates over a class data from base_component_t + 1 to component_t
335  @retval - or add class data
336  **/
337  template<callable_c<void, class_data&> callable_t = decltype(stub_callback)>
338  class_data& get_or_add_class_data(
339  const base_component_t* pRawComponent,
340  callable_t iterateClassDataFunction = stub_callback) noexcept
341  {
342  class_data* pClassData = &m_RootClass;
343  iterateClassDataFunction(*pClassData);
344 
345  std::span<const class_id> allIds = pRawComponent->get_inheritance_sequence();
346  std::span<const class_id> baseClassIds(
347  std::ranges::find(allIds, base_component_t::get_class_id_static()),
348  allIds.end());
349 
350  for (auto it = baseClassIds.begin() + 1; it != baseClassIds.end(); ++it)
351  {
352  pClassData = &pClassData->get_or_add_class_data(*it);
353  iterateClassDataFunction(*pClassData);
